Note on Quatopian API¶

Quantopian provides US equities and ETF minute data since 2002. Its API is in flux and has 3 tiers:

  • Research API : designed to be used in ipython notebook inside Quantopian platform, one can access Quantopian's by minute data

  • Open sourced zipline (and pyfolio) : almost same as Research API, but only data from web source such as Yahoo are available

  • Trading API : to be used for live trading and backtesting inside Quantopian Web IDE

This note relies on Quantopian's research API.

# two different methods of calculating cointegration

# version1 uses built in 'coint' in statsmodels.  
def find_cointegrated_pairs_v1(data):
    n = data.shape[1]
    score_matrix = np.zeros((n, n))
    pvalue_matrix_v1 = np.zeros((n, n))
    keys = data.keys()
    pairs = []
    
    for i in range(n):
        for j in range(i+1, n):
            S1 = data[keys[i]]
            S2 = data[keys[j]]
            result = coint(S1, S2, regression='nc')
            score = result[0]
            pvalue = result[1]
            score_matrix[i, j] = score
            pvalue_matrix_v1[i, j] = pvalue
            if pvalue < 0.05:
                pairs.append((keys[i], keys[j]))
                
    return pvalue_matrix_v1

# version2 uses ADF test directly, this is Moores' version
def find_cointegrated_pairs_v2(data):
    n = data.shape[1]
    adf_matrix_v2 = np.ones((n, n))
    pvalue_matrix_v2 = np.zeros((n, n))
    keys = data.keys()
    
    for i in range(n):
        for j in range(i+1, n):
            S1 = data[keys[i]]
            S2 = data[keys[j]]
            
            #res = ols(y=S2, x=S1)
            #df = S2 - res.beta.x * S1 #-res.beta.intercept
            
            # Use statsmodels directly
            S1_a = sm.add_constant(S1)
            results = sm.OLS(S2,S1_a).fit()
            
            # Moores' use no constant model, why??
            df = S2 - results.params[1]*S1 #- results.params[0]        
            cadf = ts.adfuller(df, regression='nc')
            
            adf_matrix_v2[i, j] = cadf[0]
            pvalue_matrix_v2[i, j] = cadf[1] 
                        
    return adf_matrix_v2, pvalue_matrix_v2

def heatmap(pr_df):
    p_scores_v1 = find_cointegrated_pairs_v1(pr_df)
    adf_v2, p_scores_v2 = find_cointegrated_pairs_v2(pr_df)

    fig, axes = plt.subplots(1,3)

    fig.set_size_inches(18, 5)

    axes[0].set_title("p-value using coint")

    seaborn.heatmap(p_scores_v1, xticklabels=symbol_list, yticklabels=symbol_list, cmap='RdYlGn_r' 
                , mask = (p_scores_v1 <= 0.0), ax=axes[0]
                )

    axes[1].set_title("p-value using ADF ")
    seaborn.heatmap(p_scores_v2, xticklabels=symbol_list, yticklabels=symbol_list, cmap='RdYlGn_r',
              mask = (p_scores_v2 <= 0.0), ax=axes[1] 
               )
    
    axes[2].set_title("ADF score ")
    seaborn.heatmap(adf_v2, xticklabels=symbol_list, yticklabels=symbol_list, cmap='RdYlGn_r',
              mask = (adf_v2 >= 0), ax=axes[2]
